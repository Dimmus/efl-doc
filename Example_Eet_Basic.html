22  ef = eet_open("test.eet", EET_FILE_MODE_WRITE);
23  eet_write(ef, "Entry 1", entries[0], strlen(entries[0]) + 1, 0);

28  // open the file again and blindly get the entries we wrote

31  printf("%s\n", ret);
